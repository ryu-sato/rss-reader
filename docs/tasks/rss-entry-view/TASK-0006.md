## 実装詳細

### 1. findManyEntries 関数 🔵

**信頼性**: 🔵 *要件定義 REQ-001, REQ-003, REQ-004, REQ-005・API仕様より*

```typescript
export async function findManyEntries(query: GetEntriesQuery) {
  const { feedId, tagId, page = 1, limit = 20, afterId, beforeId } = query

  const where: Prisma.EntryWhereInput = {}
  if (feedId) where.feedId = feedId
  if (tagId) where.tags = { some: { tagId } }

  // afterId/beforeId によるカーソルナビ
  if (afterId) {
    const pivot = await prisma.entry.findUnique({ where: { id: afterId } })
    if (pivot) where.publishedAt = { lt: pivot.publishedAt ?? pivot.createdAt }
  }

  const [entries, total] = await Promise.all([
    prisma.entry.findMany({
      where,
      orderBy: [{ publishedAt: 'desc' }, { createdAt: 'desc' }],
      take: limit,
      skip: afterId || beforeId ? 0 : (page - 1) * limit,
      include: { feed: true, meta: true, tags: { include: { tag: true } } },
    }),
    prisma.entry.count({ where }),
  ])

  return {
    entries,
    pagination: {
      page, limit, total,
      hasNext: total > page * limit,
      hasPrev: page > 1,
    },
  }
}
```

### 2. getEntryById 関数 🔵

**信頼性**: 🔵 *REQ-006・API仕様 GET /api/entries/:id より*

```typescript
export async function getEntryById(id: string) {
  return prisma.entry.findUnique({
    where: { id },
    include: { feed: true, meta: true, tags: { include: { tag: true } } },
  })
}
```

### 3. updateEntryMeta 関数（upsert） 🔵

**信頼性**: 🔵 *REQ-008, REQ-009, REQ-101・API仕様 PUT /api/entries/:id/meta より*

```typescript
export async function updateEntryMeta(entryId: string, data: UpdateEntryMetaInput) {
  return prisma.entryMeta.upsert({
    where: { entryId },
    create: { entryId, ...data },
    update: data,
  })
}
```
