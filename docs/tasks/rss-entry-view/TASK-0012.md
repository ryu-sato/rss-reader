## 実装詳細

### 1. Server Component でのエントリー取得 🔵

**信頼性**: 🔵 *設計文書 architecture.md ページ層・データフロー図より*

**実装ファイル**: `src/app/page.tsx`

```typescript
import { findManyEntries, getEntryById } from '@/lib/entry-service'
import { EntryList } from '@/components/entry-list'
import { EntryModal } from '@/components/entry-modal'
import { EntryFilter } from '@/components/entry-filter'

interface PageProps {
  searchParams: {
    feedId?: string
    tagId?: string
    page?: string
    entryId?: string
  }
}

export default async function Page({ searchParams }: PageProps) {
  const page = Number(searchParams.page ?? '1')
  const { entries, pagination } = await findManyEntries({
    feedId: searchParams.feedId,
    tagId: searchParams.tagId,
    page,
  })

  const selectedEntry = searchParams.entryId
    ? await getEntryById(searchParams.entryId)
    : null

  return (
    <main>
      <EntryFilter />
      <EntryList entries={entries} pagination={pagination} />
      {selectedEntry && <EntryModal entry={selectedEntry} />}
    </main>
  )
}
```
