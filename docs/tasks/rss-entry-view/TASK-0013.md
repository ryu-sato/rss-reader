## 実装詳細

### 1. エントリー一覧表示 🔵

**信頼性**: 🔵 *REQ-001, REQ-002より*

**実装ファイル**: `src/components/entry-list.tsx`

```typescript
'use client'
import { useRouter, useSearchParams } from 'next/navigation'
import { EntryListItem } from '@/types/entry'

interface EntryListProps {
  entries: EntryListItem[]
  pagination: { page: number; limit: number; total: number; hasNext: boolean; hasPrev: boolean }
}

export function EntryList({ entries, pagination }: EntryListProps) {
  const router = useRouter()
  const searchParams = useSearchParams()

  const openModal = (entryId: string) => {
    const params = new URLSearchParams(searchParams.toString())
    params.set('entryId', entryId)
    router.push(`/?${params.toString()}`)
  }

  if (entries.length === 0) {
    return <div>No entries found. Add some feeds to get started.</div>
  }

  return (
    <div>
      <ul>
        {entries.map(entry => (
          <li key={entry.id}
            onClick={() => openModal(entry.id)}
            className={entry.meta?.isRead ? 'opacity-60' : ''}
          >
            <h3>{entry.title}</h3>
            <span>{entry.feed.title}</span>
            <time>{entry.publishedAt?.toLocaleString() ?? entry.createdAt.toLocaleString()}</time>
          </li>
        ))}
      </ul>
      {/* ページネーション */}
      <div>
        <button disabled={!pagination.hasPrev}>前のページ</button>
        <span>Page {pagination.page}</span>
        <button disabled={!pagination.hasNext}>次のページ</button>
      </div>
    </div>
  )
}
```
