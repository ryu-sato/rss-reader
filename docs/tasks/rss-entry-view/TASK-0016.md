## 実装詳細

### 1. Combobox タグ入力 🔵

**信頼性**: 🔵 *設計文書 architecture.md・REQ-011, REQ-012より*

**実装ファイル**: `src/components/tag-input.tsx`

```typescript
'use client'
import { useState } from 'react'
import { Tag } from '@/types/entry'

interface TagInputProps {
  entryId: string
  initialTags: Tag[]
  allTags: Tag[]
}

export function TagInput({ entryId, initialTags, allTags }: TagInputProps) {
  const [tags, setTags] = useState(initialTags)
  const [inputValue, setInputValue] = useState('')

  const addTag = async (name: string) => {
    const res = await fetch('/api/tags', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name, entryId }),
    })
    if (res.ok) {
      const { data } = await res.json()
      setTags(prev => [...prev, data])
    }
  }

  const removeTag = async (tagId: string) => {
    const res = await fetch(`/api/tags/${tagId}/entries/${entryId}`, { method: 'DELETE' })
    if (res.ok) {
      setTags(prev => prev.filter(t => t.id !== tagId))
    }
  }

  return (
    <div>
      {/* 付与済みタグ表示 */}
      <div>
        {tags.map(tag => (
          <span key={tag.id}>
            {tag.name}
            <button onClick={() => removeTag(tag.id)}>×</button>
          </span>
        ))}
      </div>
      {/* Combobox入力 */}
    </div>
  )
}
```
